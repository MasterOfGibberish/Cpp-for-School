void Skaiciavimas (const char FR[], int a, int b, char ch, int c[])
{
   ifstream fr(FR);
  for(int i=0; i<4; i++)
  {
      fr>>a>>ch>>b;
      if(ch=='+')
      {
          c[i]=a+b;
      }  
      if(ch=='-')
      {
          c[i]=a-b;
      } 
      if(ch=='/')
      {
          c[i]=a/b;
      } 
      if(ch=='*')
      {
          c[i]=a*b;
      } 
  }
}
